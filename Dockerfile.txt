# Etapa 1 — build da aplicação Vite
FROM node:20-alpine AS builder
WORKDIR /app

# Copia package.json e lockfile
COPY package*.json ./
COPY pnpm-lock.yaml* ./

# Instala pnpm e dependências
RUN npm install -g pnpm && \
    pnpm install --frozen-lockfile

# Copia código e faz build
COPY . .
RUN pnpm build

# Etapa 2 — servidor Nginx
FROM nginx:alpine

# Instala envsubst para processar variáveis
RUN apk add --no-cache gettext

# Copia arquivos estáticos do Vite
COPY --from=builder /app/dist /usr/share/nginx/html

# Copia template do Nginx
COPY nginx.conf.template /etc/nginx/templates/default.conf.template

# Expõe porta 8080
EXPOSE 8080

# Processa template e inicia Nginx
CMD ["/bin/sh", "-c", "envsubst '${PORT}' < /etc/nginx/templates/default.conf.template > /etc/nginx/conf.d/default.conf && exec nginx -g 'daemon off;'"]
